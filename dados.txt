 num = input("Digite o Nº de estudante: ").strip()

    try:
        with open("aluno1.txt", "r", encoding="utf-8") as f:
            linhas = f.readlines()
    except FileNotFoundError:
        print("Arquivo não encontrado.")
        return

    blocos = []
    bloco_atual = []

    for linha in linhas:
        if linha.strip() == "":
            if bloco_atual:
                blocos.append(bloco_atual)
                bloco_atual = []
        else:
            bloco_atual.append(linha)
    if bloco_atual:
        blocos.append(bloco_atual)

    encontrou = False
    for i, bloco in enumerate(blocos):
        if any(f"Nº estudante: {num}" in linha for linha in bloco):
            encontrou = True
            print("\nBloco encontrado:")
            print("".join(bloco))
            print("\nO que deseja retificar?")
            print("1 - Nome")
            print("2 - Nota1")
            print("3 - Nota2")
            print("4 - Todo o bloco")
            opcao = input("Escolha uma opção: ").strip()

            if opcao == "1":
                novo_nome = input("Novo nome: ").strip().title()
                for j in range(len(bloco)):
                    if bloco[j].startswith("Nome:"):
                        bloco[j] = f"Nome: {novo_nome}\n"
                        break

            elif opcao == "2":
                nova_nota1 = input("Nova Nota1: ").strip()
                for j in range(len(bloco)):
                    if bloco[j].startswith("Nota1:"):
                        bloco[j] = f"Nota1: {nova_nota1}\n"
                        break

            elif opcao == "3":
                nova_nota2 = input("Nova Nota2: ").strip()
                for j in range(len(bloco)):
                    if bloco[j].startswith("Nota2:"):
                        bloco[j] = f"Nota2: {nova_nota2}\n"
                        break

            elif opcao == "4":
                print("\nDigite o novo bloco completo (termine com linha vazia):")
                novo_bloco = []
                while True:
                    linha_nova = input()
                    if linha_nova == "":
                        break
                    novo_bloco.append(linha_nova + "\n")
                    bloco = novo_bloco

            blocos[i] = bloco
            break

    if not encontrou:
        print("Aluno não encontrado.")
        return

    with open("aluno1.txt", "w", encoding="utf-8") as f:
        for bloco in blocos:
            for linha in bloco:
                f.write(linha)
            f.write("\n")

    print("\n\033[1;32mAlteração feita com sucesso!\033[m")
